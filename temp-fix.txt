    /**
     * Send the notification via FirebaseMessaging.
     *
     * @param mixed $notifiable
     * @return void
     */
    public function toFirebase($notifiable)
    {
        // Get device token from notifiable (e.g. User model)
        $deviceToken = $notifiable->device_token ?? null;

        // If no device token, do not send notification
        if (!$deviceToken) {
            return;
        }

        // Send the notification via FirebaseService
        $this->firebaseService->sendToDevice($deviceToken, $this->title, $this->body, $this->data);
    }
